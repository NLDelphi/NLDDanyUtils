Example of the usage of the "NLDMailOut" unit



// inclusion in the uses clause of your program

uses
  ..., NLDMailOut, ...;



// This is the procedure to send the mail

procedure SendMailNow;
var List: TStrings;                // only needed if you want to send attachments
begin
  
  // Prepare for attachments (if any) 
  List := TStringList.Create;      // only needed if you want to send attachments
  List.Add('test.html');           // only needed if you want to send attachments, do this for
  List.Add('...');                 //   every attchment	

  // Overrule default settings if necessary.
  // Default the Mail server parameters and the addresses set with "SMTPSetup" are used, but
  // If you want to overrule the mail parameters set in "SMTPSetup" then do the following:
  //   to overrule the mail server parameters:
  SetMailOutParams(Host, Port, AuthenticationType, Id, Pw);
  //   to overrule the sender mail address and the return mail address:
  SetMailOutAddresses(From, Answer);
  

  // Send the mail
  SendMail('Subject',              // the subject of your mail

    'aaa.bbb@ccc',                 // the destination(s) of your mail, comma separated.

    'This is the body of my mail', // the body of your mail, can be a single string, or
                                   //   a TStrings type

    '',                            // the 'CC' list of your mail, comma separated.
                                   //   if you have no 'CC' or 'BCC'list and no attachments
                                   //   you can leave this parameter and also the rest out 
 
    '',                            // the 'BCC' list of your mail, comma separated.
                                   //   if you have no 'BCC'list and no attachments
                                   //   you can leave this parameter and also the rest out

    List                           // The list of attachments (can be left out if none)
  );

  // Clean up 
  List.Free;                       // only needed if you want to send attachments
  
end;



// that's all folks!
